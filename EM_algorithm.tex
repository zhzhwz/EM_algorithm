    \begin{itemize}
        \item $X \sim B(10, \Gh_1)$
        \item $X_1 = 9, X_2 = 8, X_3 = 7$
        \item 极大似然估计为 $\hat{\Gh_1} = \frac{\overline{X}}{10} = 0.8$
    \end{itemize}
